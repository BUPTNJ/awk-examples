BEGIN{
	recent="none"
	count=0
}
{
	match($0,/[0-9]{4}-[0-9]{2}-[0-9]{2}/,tim);
	if(tim[0]!=""){
		day=tim[0];
	}
	match($0,/[0-9]+: ?[0-9]+/,clock);
	if(clock[0]!=""){
		ck[count]=clock[0];
		count++;
	}	
	match($0,/<td class="t2">([^0-9]*)<\/td>/,a);
	if (a[1]!=""){
		recent=a[1]
		count=0
	}
	match($0,/<td class="t2_1">çº¦(.*)<\/td>/,c);
	if(c[1]!=""){
		printf("%s,%s %s,%s\n",recent,day,ck[count],c[1])
		count++ 
	}
}
